def permutation(list):
    if len(list)==0:
        return 0
    if len(list)==1:
        return [list]
    l=[]
    for i in range(len(list)):
        m=list[i]
        rem=list[:i]+list[i+1:]
        for j in permutation(rem):
            l.append([m]+j)
    return l
li=[1,2,3]
for p in permutation(li):
    print(p)
